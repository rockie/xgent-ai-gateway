pub struct RegisterServiceRequest {
    pub name: String,                        // required

    pub max_retries: Option<u32>,            // optional, server default
    pub max_nodes: Option<u32>,              // optional, server default
    pub node_stale_after_secs: Option<u64>,  // optional, server default
    pub drain_timeout_secs: Option<u64>,     // optional, server default
}

// GET /v1/admin/services response item; also returned from POST (201)
pub struct ServiceResponse {
    pub name: String,
    pub description: String,
    pub created_at: String,           // RFC3339
    pub task_timeout_secs: u64,
    pub max_retries: u32,
    pub max_nodes: Option<u32>,       // null = unlimited
    pub node_stale_after_secs: u64,
    pub drain_timeout_secs: u64,
}

// GET /v1/admin/services response wrapper
pub struct ListServicesResponse {
    pub services: Vec<ServiceResponse>,
}

// GET /v1/admin/services/{name} response (flattened ServiceResponse + nodes)
pub struct ServiceDetailResponse {
    #[serde(flatten)]
    pub service: ServiceResponse,     // all ServiceResponse fields at top level
    pub nodes: Vec<NodeStatusResponse>,
}

pub struct NodeStatusResponse {
    pub node_id: String,
    pub health: String,       // "healthy" | "unhealthy" | "disconnected"
    pub last_seen: String,    // RFC3339
    pub in_flight_tasks: u32,
    pub draining: bool,
}

// DELETE /v1/admin/services/{name} returns 202 Accepted (no body)

' + (error as Error).message)`.
- `useDeregisterService()`: useMutation calling `apiClient<void>('/v1/admin/services/${encodeURIComponent(name)}', { method: 'DELETE' })`. On success: `queryClient.invalidateQueries({ queryKey: ['services'] })` and `toast.success('Service deregistration started')`. On error: `toast.error('Failed to deregister service. ' + (error as Error).message)`.

Utility function:
```typescript
export function deriveServiceHealth(nodes: NodeStatusResponse[]): 'healthy' | 'degraded' | 'down' | 'unknown' {
  if (nodes.length === 0) return 'unknown'
  const healthyCount = nodes.filter(n => n.health === 'healthy' && !n.draining).length
  if (healthyCount === nodes.length) return 'healthy'
  if (healthyCount === 0) return 'down'
  return 'degraded'
}
```

Relative time utility:
```typescript
export function relativeTime(isoString: string): string {
  const diffMs = Date.now() - new Date(isoString).getTime()
  if (diffMs < 0) return 'just now'
  const seconds = Math.floor(diffMs / 1000)
  if (seconds < 60) return `${seconds}s ago`
  const minutes = Math.floor(seconds / 60)
  if (minutes < 60) return `${minutes}m ago`
  const hours = Math.floor(minutes / 60)
  if (hours < 24) return `${hours}h ago`
  return `${Math.floor(hours / 24)}d ago`
}
```
