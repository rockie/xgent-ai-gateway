       - pub struct AgentConfig { pub gateway: GatewaySection, pub service: ServiceSection, pub cli: Option<CliSection>, pub response: ResponseSection }
       - pub struct GatewaySection { pub address: String, pub token: String, pub node_id: String (default uuid v7), pub ca_cert: Option<String>, pub tls_skip_verify: bool (default false), pub max_reconnect_delay_secs: u64 (default 30) }

       - pub enum CliInputMode { Arg, Stdin } with #[serde(rename_all = "lowercase")]
       - pub struct ResponseSection { pub body: String, pub max_bytes: usize (default 1_048_576) }
       - pub fn load_config(path: &str) -> Result<AgentConfig, String>:
         (a) Read file to string
         (b) Call interpolate_env_vars(&raw) to resolve ${VAR} patterns
         (c) Parse with serde_yaml_ng::from_str
         (d) Validate: if mode is Cli, cli section must be present
       - fn interpolate_env_vars(raw: &str) -> Result<String, String>:
         Scan for "${" ... "}" pattern. For each match, look up std::env::var(name).
         If missing, return Err("missing environment variable: ${NAME}").
         Replace all occurrences. Use manual string scanning (no regex dependency needed).
       - #[cfg(test)] mod tests with tests for all behaviors listed above

    6. Create gateway/src/agent/placeholder.rs (per D-05, D-06, D-08, D-09):
       - pub fn resolve_placeholders(template: &str, variables: &HashMap<String, String>) -> Result<String, String>
       - Character-scanning implementation: iterate chars, on '<' scan for '>', look up token in variables map.
         If token not found, return Err with message "unresolved placeholder <TOKEN>; available: [key1, key2, ...]"
         If no closing '>', preserve literal '<' and scanned chars.
         Resolved values are pushed to output buffer and never re-scanned (single-pass, D-09).
       - pub fn build_task_variables(assignment: &TaskAssignment, service_name: &str) -> HashMap<String, String>:
         Insert "payload" => String::from_utf8_lossy(&assignment.payload),
         Insert "service_name" => service_name,
         For each (k, v) in assignment.metadata: insert "metadata.{k}" => v
       - #[cfg(test)] mod tests with tests for all behaviors listed above
