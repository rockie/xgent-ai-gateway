    pub struct CliExecutor:
      - Fields: service_name: String, cli: CliSection, response: ResponseSection
      - Constructor: pub fn new(service_name: String, cli: CliSection, response: ResponseSection) -> Self

    #[async_trait] impl Executor for CliExecutor:
      async fn execute(&self, assignment: &TaskAssignment) -> ExecutionResult:
        (a) Build task variables via placeholder::build_task_variables(assignment, &self.service_name)
        (b) Resolve placeholders in each element of self.cli.command using resolve_placeholders
            - Build resolved_command: Vec<String> by mapping each element through resolve_placeholders
        (c) Match on self.cli.input_mode:
            - Arg mode: spawn process with resolved_command[0] as program, resolved_command[1..] as args
              Stdout and stderr piped, stdin null. kill_on_drop(true).
            - Stdin mode: spawn with original (unresolved for payload? No — resolve all non-payload placeholders
              in command, then pipe raw payload bytes to stdin).
              Actually: resolve ALL placeholders in command (including <payload> if present in command elements).
              Stdin receives assignment.payload as raw bytes.
              stdin, stdout, stderr all piped. kill_on_drop(true).
        (d) If self.cli.cwd is Some(dir), set .current_dir(dir) on Command
        (e) For each (k,v) in self.cli.env, set .env(k, v) on Command
        (f) Stdin mode concurrent I/O (per D-11):
            - Take stdin handle, spawn tokio task to write payload bytes then shutdown stdin
            - Take stdout handle, spawn tokio task to read_to_end
            - Take stderr handle, spawn tokio task to read_to_end
        (g) Arg mode: just take stdout and stderr handles, spawn read tasks
        (h) Wrap child.wait() in tokio::time::timeout(Duration::from_secs(self.cli.timeout_secs))
            On timeout: explicitly call child.kill().await (per D-12, RESEARCH Pitfall 2).
            Return ExecutionResult { success: false, result: vec![], error_message: format!("process timed out after {} seconds", self.cli.timeout_secs) }
        (i) Collect stdout and stderr from spawned tasks
        (j) Check exit code: status.code().unwrap_or(-1)
            If exit_code != 0: return ExecutionResult { success: false, result: vec![], error_message: format!("process exited with code {}", exit_code) } (per D-14)
        (k) Add "stdout" and "stderr" to task variables map
        (l) Call response::resolve_response_body(&self.response.body, &variables, self.response.max_bytes)
            On Ok(bytes): return ExecutionResult { success: true, result: bytes, error_message: String::new() }
            On Err(e): return ExecutionResult { success: false, result: vec![], error_message: e }
